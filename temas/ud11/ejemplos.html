💾 Ejemplos con Ficheros Binarios en Java

Guardar 5 números enteros en un fichero binario.


import java.io.*;

public class Ej1 {
    public static void main(String[] args) throws IOException {
        DataOutputStream dos = new DataOutputStream(new FileOutputStream("numeros.dat"));

        for(int i=1;i<=5;i++){
            dos.writeInt(i*10);
        }

        dos.close();
    }
}



import java.io.*;

public class Ej2 {
    public static void main(String[] args) throws IOException {
        DataInputStream dis = new DataInputStream(new FileInputStream("numeros.dat"));

        try {
            while(true){
                System.out.println(dis.readInt());
            }
        } catch(EOFException e){
            System.out.println("Fin del fichero");
        }

        dis.close();
    }
}

import java.io.*;

class Persona implements Serializable {
    String nombre;
    int edad;

    Persona(String n, int e){
        nombre = n;
        edad = e;
    }
}

public class Ej4 {
    public static void main(String[] args) throws IOException {
        ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream("persona.dat"));

        Persona p = new Persona("Juan", 30);
        oos.writeObject(p);

        oos.close();
    }
}

import java.io.*;

class Alumno implements Serializable {
    int id;
    String nombre;
    double nota;

    Alumno(int id, String nombre, double nota){
        this.id = id;
        this.nombre = nombre;
        this.nota = nota;
    }
}

public class AltaAlumno {
    public static void main(String[] args) throws Exception {
        ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream("alumnos.dat"));

        oos.writeObject(new Alumno(1,"Ana",7.5));
        oos.writeObject(new Alumno(2,"Luis",8.2));

        oos.close();
    }
}



import java.io.*;

public class Listar {
    public static void main(String[] args) throws Exception {
        ObjectInputStream ois = new ObjectInputStream(new FileInputStream("alumnos.dat"));

        try{
            while(true){
                Alumno a = (Alumno) ois.readObject();
                System.out.println(a.id + " " + a.nombre + " " + a.nota);
            }
        } catch(EOFException e){
            System.out.println("Fin");
        }

        ois.close();
    }
}



import java.io.*;

public class Buscar {
    public static void main(String[] args) throws Exception {
        int buscar = 2;

        ObjectInputStream ois = new ObjectInputStream(new FileInputStream("alumnos.dat"));

        try{
            while(true){
                Alumno a = (Alumno) ois.readObject();
                if(a.id == buscar){
                    System.out.println("Encontrado: " + a.nombre);
                }
            }
        } catch(EOFException e){}

        ois.close();
    }
}



import java.io.*;

public class Modificar {
    public static void main(String[] args) throws Exception {

        ObjectInputStream ois = new ObjectInputStream(new FileInputStream("alumnos.dat"));
        ObjectOutputStream oos = new ObjectOutputStream(new FileOutputStream("temp.dat"));

        try{
            while(true){
                Alumno a = (Alumno) ois.readObject();

                if(a.id == 2){
                    a.nota = 10;
                }

                oos.writeObject(a);
            }
        } catch(EOFException e){}

        ois.close();
        oos.close();

        new File("alumnos.dat").delete();
        new File("temp.dat").renameTo(new File("alumnos.dat"));
    }
}
