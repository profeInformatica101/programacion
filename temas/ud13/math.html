operaciones agregadas · filter + reduce + sum 
import java.util.*;
import java.util.stream.*;

class Transaccion {
    String categoria;
    double monto;
    Transaccion(String c, double m) { categoria = c; monto = m; }
    public String getCategoria() { return categoria; }
    public double getMonto() { return monto; }
}

List<Transaccion> transacciones = Arrays.asList(
    new Transaccion("comida", 45.50),
    new Transaccion("transporte", 12.30),
    new Transaccion("comida", 32.00),
    new Transaccion("ocio", 100.00),
    new Transaccion("comida", 18.75)
);

// 1. Filtrar transacciones de comida con monto > 20 y sumar montos
double sumaComidaFiltrada = transacciones.stream()
    .filter(t -> t.getCategoria().equals("comida"))
    .filter(t -> t.getMonto() > 20)
    .mapToDouble(Transaccion::getMonto)
    .sum();

// 2. Reducción personalizada: obtener categorías únicas mediante reduce (Set)
Set<String> categoriasUnicas = transacciones.stream()
    .map(Transaccion::getCategoria)
    .reduce(new HashSet<>(), (set, cat) -> { set.add(cat); return set; }, (a,b) -> { a.addAll(b); return a; });

System.out.println("Suma comidas >20€: " + sumaComidaFiltrada);
System.out.println("Categorías únicas: " + categoriasUnicas);
▶️ Salida esperada:
Suma comidas >20€: 77.5
Categorías únicas: [comida, transporte, ocio]
⚡ Agregación: filter + mapToDouble + sum / reduce con acumulador mutable.

Collectors.toMap · groupingBy · summarizing 
record Empleado(String nombre, String depto, double salario) {}

List<Empleado> empleados = List.of(
    new Empleado("Ana", "IT", 45000),
    new Empleado("Luis", "IT", 52000),
    new Empleado("Marta", "RRHH", 38000),
    new Empleado("Carlos", "RRHH", 41000),
    new Empleado("Eva", "IT", 49000)
);

// GroupingBy + promedio por departamento
Map<String, Double> salarioPromedioPorDepto = empleados.stream()
    .collect(Collectors.groupingBy(Empleado::depto,
             Collectors.averagingDouble(Empleado::salario)));

// PartitioningBy: separar salarios altos (>45000)
Map<Boolean, List<Empleado>> particionSalario = empleados.stream()
    .collect(Collectors.partitioningBy(e -> e.salario() > 45000));

// SummarizingDouble: estadísticas completas
DoubleSummaryStatistics stats = empleados.stream()
    .mapToDouble(Empleado::salario)
    .summaryStatistics();

System.out.println("Promedio por depto: " + salarioPromedioPorDepto);
System.out.println("Salario >45k: " + particionSalario.get(true).stream().map(Empleado::nombre).toList());
System.out.println("Estadísticas salariales: max=" + stats.getMax() + ", avg=" + stats.getAverage());

flatMap · sorted(Comparator) · peek debug 
// Estructura anidada: cada equipo tiene lista de integrantes
record Equipo(String nombreEquipo, List<String> miembros) {}

List<Equipo> equipos = List.of(
    new Equipo("Alpha", List.of("Juan", "Maria", "Pedro")),
    new Equipo("Beta", List.of("Lucia", "Andres")),
    new Equipo("Gamma", List.of("Sofia", "Mateo", "Valentina"))
);

// FlatMap para obtener todos los miembros + orden alfabético inverso + filtrar longitud >4
List<String> miembrosFiltradosOrdenados = equipos.stream()
    .flatMap(equipo -> equipo.miembros().stream())
    .filter(m -> m.length() > 4)
    .sorted((a,b) -> b.compareTo(a))   // descendente
    .peek(System.out::println)          // operación intermedia para depurar
    .toList();

// También transformación: nombre -> mayúscula + contar caracteres
List<String> reporte = equipos.stream()
    .flatMap(e -> e.miembros().stream())
    .map(m -> m.toUpperCase() + " (" + m.length() + ")")
    .toList();

System.out.println("Miembros filtrados (longitud>4) descendente: " + miembrosFiltradosOrdenados);
System.out.println("Reporte transformado: " + reporte);

Simulación JSON · Jackson-like con Streams 
// Simulación de datos provenientes de JSON (Lista de Mapas)
List<Map<String, Object>> jsonData = List.of(
    Map.of("id", 1, "nombre", "Laptop", "precio", 1200.0, "tags", List.of("electronica", "oferta")),
    Map.of("id", 2, "nombre", "Mouse", "precio", 25.5, "tags", List.of("accesorio", "basico")),
    Map.of("id", 3, "nombre", "Monitor", "precio", 350.0, "tags", List.of("electronica", "gaming"))
);

// Operaciones: filtrar productos con precio > 100 y extraer nombre + precio, ordenar por precio desc
List<String> reporteProductos = jsonData.stream()
    .filter(item -> ((Number) item.get("precio")).doubleValue() > 100)
    .sorted((a,b) -> Double.compare(
        ((Number) b.get("precio")).doubleValue(),
        ((Number) a.get("precio")).doubleValue()))
    .map(item -> item.get("nombre") + " - $" + item.get("precio"))
    .toList();

// Uso de flatMap para extraer todos los tags únicos
Set<String> tagsUnicos = jsonData.stream()
    .flatMap(item -> ((List<String>) item.get("tags")).stream())
    .collect(Collectors.toSet());

System.out.println("Productos caros ordenados: " + reporteProductos);
System.out.println("Tags únicos desde JSON: " + tagsUnicos);
▶️ Salida JSON-like procesada:
Productos caros ordenados: [Laptop - $1200.0, Monitor - $350.0]
Tags únicos: [electronica, oferta, accesorio, basico, gaming]
Simula procesamiento de flujos JSON usando Java Streams sin librerías externas, ideal para ETL ligero.

modelado de documento XML → Stream processing 
// Representación de un documento XML simple con objetos anidados
record Libro(String titulo, String autor, int anyo, double precio) {}
record Libreria(String nombre, List<Libro> libros) {}

Libreria libreria = new Libreria("Biblioteca Central", List.of(
    new Libro("Effective Java", "Joshua Bloch", 2018, 45.0),
    new Libro("Java Concurrency", "Brian Goetz", 2015, 55.0),
    new Libro("Modern Java", "Raoul-Gabriel Urma", 2021, 39.99),
    new Libro("Clean Code", "Robert C. Martin", 2008, 48.50)
));

// Usando streams para extraer información como si se procesara XML (xpath-like)
List<String> librosRecientes = libreria.libros().stream()
    .filter(libro -> libro.anyo() >= 2018)
    .sorted((a,b) -> Integer.compare(b.anyo(), a.anyo()))
    .map(libro -> "📖 " + libro.titulo() + " (" + libro.anyo() + ") - $" + libro.precio())
    .toList();

// Reducción: precio total de libros con descuento simulado (recolección)
double sumaPrecios = libreria.libros().stream()
    .mapToDouble(Libro::precio)
    .sum();

// Agrupar por rango de año (clásica agregación)
Map<String, List<Libro>> librosPorDecada = libreria.libros().stream()
    .collect(Collectors.groupingBy(l -> (l.anyo() / 10) * 10 + "s"));

System.out.println("📌 XML procesado (Libros recientes): " + librosRecientes);
System.out.println("💰 Precio total colección: $" + sumaPrecios);
System.out.println("📚 Clasificación por década: " + librosPorDecada.keySet());
▶️ Salida XML/Streams:
Libros recientes: [📖 Modern Java (2021) - $39.99, 📖 Effective Java (2018) - $45.0]
Precio total colección: $188.49
Clasificación por década: [2010s, 2000s, 2020s]

// Collector personalizado para calcular media y desviación (simplificado)
record Estadisticas(double media, double varianza) {}

List<Double> muestras = List.of(10.0, 20.0, 30.0, 40.0, 50.0);

Estadisticas statsPersonalizadas = muestras.stream()
    .collect(Collector.of(
        () -> new double[]{0, 0, 0}, // [sum, sumSq, count]
        (acc, val) -> { acc[0] += val; acc[1] += val*val; acc[2]++; },
        (a, b) -> { a[0] += b[0]; a[1] += b[1]; a[2] += b[2]; return a; },
        acc -> {
            double media = acc[0] / acc[2];
            double varianza = (acc[1] / acc[2]) - (media * media);
            return new Estadisticas(media, varianza);
        }
    ));

System.out.printf("Media = %.2f , Varianza = %.4f%n", statsPersonalizadas.media(), statsPersonalizadas.varianza());
Media = 30.00 , Varianza = 200.0000
