Objetivo didáctico
Implementar ordenaciones reales en un dominio concreto, entendiendo cuándo usar Comparable (orden natural) y cuándo usar Comparator (criterios alternativos), con desempates y datos nulos.
Entrega
Repositorio con /src + README.md + /test (JUnit).

Situación-problema

Un marketplace educativo vende Productos. Debes implementar una ordenación por defecto (orden natural) y varias ordenaciones alternativas (comparators), trabajando con tablas (arrays).

Qué se pide
  • Comparable: un orden natural único.
  • Comparator: mínimo 4 criterios alternativos.
  • Catálogo: operaciones con tablas.
  • Pruebas: varios tests unitarios.
Casos borde
  • Empates (mismo precio / rating / ventas).
  • categoria puede ser null.
  • Desempates estables (sku).

Orden natural (Comparable)
  • ventas desc
  • si empatan: rating desc
  • si empatan: nombre asc (ignoreCase)
  • si empatan: sku asc
public class Producto implements Comparable<Producto> {

  @Override
  public int compareTo(Producto other) {
    int c = Integer.compare(other.getVentas(), this.getVentas());
    if (c != 0) return c;

    c = Double.compare(other.getRating(), this.getRating());
    if (c != 0) return c;

    c = this.getNombre().compareToIgnoreCase(other.getNombre());
    if (c != 0) return c;

    return this.getSku().compareToIgnoreCase(other.getSku());
  }
}

  • Precio asc (empate: nombre asc)
  • Precio desc (empate: nombre asc)
  • Stock asc (empate: ventas desc)
  • Categoría asc (nulls last) y dentro: rating desc
import java.util.Comparator;

public final class ComparadoresProducto {

  private ComparadoresProducto() {}

  public static final Comparator<Producto> POR_PRECIO_ASC =
      Comparator.comparingDouble(Producto::getPrecio)
                .thenComparing(p -> p.getNombre().toLowerCase());

  public static final Comparator<Producto> POR_PRECIO_DESC =
      POR_PRECIO_ASC.reversed();

  public static final Comparator<Producto> POR_STOCK_ASC =
      Comparator.comparingInt(Producto::getStock)
                .thenComparing(Comparator.comparingInt(Producto::getVentas).reversed());

  public static final Comparator<Producto> POR_CATEGORIA_Y_RATING =
      Comparator.comparing(Producto::getCategoria,
                  Comparator.nullsLast(String.CASE_INSENSITIVE_ORDER))
                .thenComparing(Comparator.comparingDouble(Producto::getRating).reversed());
}

Importante
El catálogo se guarda en Producto[]. Para ordenar, trabaja sobre una copia. La ordenación será manual (burbuja), usando compareTo o Comparator. 
import java.util.Arrays;
import java.util.Comparator;
import java.util.Objects;

public class CatalogoArray {

  private final Producto[] productos;

  public CatalogoArray(Producto[] productos) {
    this.productos = Arrays.copyOf(productos, productos.length);
  }

  public Producto[] copiar() {
    return Arrays.copyOf(productos, productos.length);
  }

  public void ordenarOrdenNatural(Producto[] t) {
    for (int i = 0; i < t.length - 1; i++) {
      for (int j = 0; j < t.length - 1 - i; j++) {
        if (t[j].compareTo(t[j + 1]) > 0) {
          Producto tmp = t[j];
          t[j] = t[j + 1];
          t[j + 1] = tmp;
        }
      }
    }
  }

  public void ordenar(Producto[] t, Comparator<Producto> c) {
    for (int i = 0; i < t.length - 1; i++) {
      for (int j = 0; j < t.length - 1 - i; j++) {
        if (c.compare(t[j], t[j + 1]) > 0) {
          Producto tmp = t[j];
          t[j] = t[j + 1];
          t[j + 1] = tmp;
        }
      }
    }
  }

  public Producto[] filtrarPorCategoria(String cat) {
    int count = 0;
    for (Producto p : productos) {
      if (Objects.equals(p.getCategoria(), cat)) count++;
    }

    Producto[] out = new Producto[count];
    int k = 0;
    for (Producto p : productos) {
      if (Objects.equals(p.getCategoria(), cat)) {
        out[k++] = p;
      }
    }
    return out;
  }

  public Producto[] topN(int n, Comparator<Producto> c) {
    Producto[] copia = copiar();
    ordenar(copia, c);
    int m = Math.min(n, copia.length);
    return Arrays.copyOf(copia, m);
  }
}

import org.junit.jupiter.api.Test;

import java.util.Comparator;

import static org.junit.jupiter.api.Assertions.*;

class CatalogoArrayTest {

  private static Producto p(String sku, String nombre, String cat,
                            double precio, double rating, int stock, int ventas) {
    return new Producto(sku, nombre, cat, precio, rating, stock, ventas);
  }

  private CatalogoArray catalogoBase() {
    Producto[] base = new Producto[] {
        p("LIB-001","Clean Code","Libros",34.95,4.7,3,210),
        p("LIB-014","Effective Java","Libros",45.50,4.9,8,320),
        p("HWD-210","Teclado Mecánico","Hardware",59.99,4.3,12,180),
        p("LIC-777","IDE Pro (1 año)","Licencias",89.00,4.6,50,95),
        p("HWD-011","Ratón Vertical","Hardware",29.90,4.2,2,140),
        p("LIB-002","Refactoring","Libros",41.00,4.8,1,210),
        p("ACC-100","Mochila Portátil",null,24.99,4.1,20,60),
        p("HWD-099","Monitor 27"","Hardware",159.00,4.5,4,75),
        p("LIC-120","Antivirus EDU","Licencias",19.99,3.9,100,500),
        p("LIB-099","Java Concurrency","Libros",39.90,4.4,6,120),
        p("LIB-050","Algoritmos","Libros",39.90,4.4,6,120),
        p("HWD-500","SSD 1TB","Hardware",89.00,4.6,0,95)
    };
    return new CatalogoArray(base);
  }

  @Test
  void ordenarOrdenNatural_debePonerPrimero_elDeMasVentas() {
    CatalogoArray c = catalogoBase();
    Producto[] t = c.copiar();
    c.ordenarOrdenNatural(t);

    assertEquals("LIC-120", t[0].getSku(), "El de más ventas (500) debe ir primero");
  }

  @Test
  void ordenarOrdenNatural_siEmpatanVentas_debeUsarRatingDesc() {
    // Ventas empatan a 210: Clean Code (4.7) vs Refactoring (4.8)
    CatalogoArray c = new CatalogoArray(new Producto[] {
        p("LIB-001","Clean Code","Libros",34.95,4.7,3,210),
        p("LIB-002","Refactoring","Libros",41.00,4.8,1,210)
    });

    Producto[] t = c.copiar();
    c.ordenarOrdenNatural(t);

    assertEquals("LIB-002", t[0].getSku(), "Con mismas ventas, mayor rating primero");
  }

  @Test
  void ordenarOrdenNatural_siEmpatanVentasYRating_debeUsarNombreAsc() {
    // Ventas 120 y rating 4.4: "Algoritmos" debe ir antes que "Java Concurrency"
    CatalogoArray c = new CatalogoArray(new Producto[] {
        p("LIB-099","Java Concurrency","Libros",39.90,4.4,6,120),
        p("LIB-050","Algoritmos","Libros",39.90,4.4,6,120)
    });

    Producto[] t = c.copiar();
    c.ordenarOrdenNatural(t);

    assertEquals("LIB-050", t[0].getSku(), "Con empate, nombre A-Z");
  }

  @Test
  void ordenarPrecioDesc_enLibros_debeDejarPrimero_EffectiveJava() {
    CatalogoArray c = catalogoBase();
    Producto[] libros = c.filtrarPorCategoria("Libros");

    // Comparator equivalente al pedido (precio desc + nombre asc)
    Comparator<Producto> porPrecioDesc =
        Comparator.comparingDouble(Producto::getPrecio).reversed()
                  .thenComparing(p -> p.getNombre().toLowerCase());

    c.ordenar(libros, porPrecioDesc);

    assertEquals("LIB-014", libros[0].getSku(), "En Libros, el más caro debe ser Effective Java");
  }

  @Test
  void ordenarCategoria_nullsLast_debeDejarNullAlFinal() {
    CatalogoArray c = new CatalogoArray(new Producto[] {
        p("A","A",null,1,1,1,1),
        p("B","B","Hardware",1,1,1,1),
        p("C","C","Libros",1,1,1,1)
    });

    Producto[] t = c.copiar();

    Comparator<Producto> porCategoriaNullsLast =
        Comparator.comparing(Producto::getCategoria,
            Comparator.nullsLast(String.CASE_INSENSITIVE_ORDER));

    c.ordenar(t, porCategoriaNullsLast);

    assertNull(t[t.length - 1].getCategoria(), "La categoría null debe quedar al final");
  }

  @Test
  void topN_debeDevolverLongitudN_yOrdenado() {
    CatalogoArray c = catalogoBase();

    Comparator<Producto> porStockAsc =
        Comparator.comparingInt(Producto::getStock)
                  .thenComparing(Comparator.comparingInt(Producto::getVentas).reversed());

    Producto[] top3 = c.topN(3, porStockAsc);

    assertEquals(3, top3.length);
    assertTrue(top3[0].getStock() <= top3[1].getStock());
    assertTrue(top3[1].getStock() <= top3[2].getStock());
  }
}
